# 189. Rotate Array

Reverse:

```
class Solution {
    public void rotate(int[] nums, int k) {
        k %= nums.length;
        reverse(nums, 0, nums.length - 1);
        reverse(nums, 0, k - 1);
        reverse(nums, k, nums.length - 1);
    }
    public void reverse(int[] nums, int l, int r){
        while(l < r){
            int temp = nums[l];
            nums[l] = nums[r];
            nums[r] = temp;
            l++;
            r--;
        }
        
    }
}
```

```
class Solution {
    public void rotate(int[] nums, int k) {
        int n = nums.length;
        k %= n;
        
        int [] res = new int [n];
        
        for(int i = 0; i < nums.length; i++)
        {
            res[(k + i) % nums.length] = nums[i];
        }
        
        int x = 0;
        for(int num : res)
            nums[x++] = num;
    }
}
```
