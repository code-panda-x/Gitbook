# 234. Palindrome Linked List

```
class Solution {
   
    public boolean isPalindrome(ListNode head) {
        if(head == null)
            return false;
        
        ListNode slow = head;
        ListNode fast = head;
        
        while(fast != null && fast.next != null){
            slow = slow.next;
            fast = fast.next.next;
        }
        if(fast != null)
            slow = slow.next;
        
        slow = reverse(slow);
        ListNode newcur = head;
        
        while(slow != null){
            if(newcur.val != slow.val)
                return false;
            slow = slow.next;
            newcur = newcur.next;
        }
        return true;
        
    }
    public ListNode reverse(ListNode head){
        ListNode prev = null;
        
        while(head != null){
            ListNode next = head.next;
            head.next = prev;
            prev = head;
            head = next;
        }
        return prev;
    }
   
}
```
