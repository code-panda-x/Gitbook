# 209. Minimum Size Subarray Sum

```
class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        if(nums.length == 0 || nums == null)
            return 0;
        int res = Integer.MAX_VALUE;
        int i = 0, j = 0;
        int sum = 0;
        
        while(j < nums.length){
            sum += nums[j++];
            while(sum >= target){
                res = Math.min(res, j-i);
                sum -= nums[i++];
            }
        }
        return res == Integer.MAX_VALUE ? 0 : res;
    }
}
```
