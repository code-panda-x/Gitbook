# 1248. Count Number of Nice Subarrays

at most

```
class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        return atMost(nums, k) - atMost(nums, k-1);
    }
    public int atMost(int[] nums, int k){
        int res = 0;
        int l = 0;
        int r = 0;
        
        while(r < nums.length){
            if(nums[r] % 2 == 1)
                k--;
            r++;
            
            while(k < 0){
                if(nums[l] % 2 == 1)
                    k++;
                l++;
            }
            res += r - l + 1;          
        }
        return res;
    }
}
```

```
        while(j < A.length) {
            if (A[j] % 2 == 1) {
                k--;
                count = 0;
```
