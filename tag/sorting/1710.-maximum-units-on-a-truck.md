# 1710. Maximum Units on a Truck

```
class Solution {
    public int maximumUnits(int[][] boxTypes, int truckSize) {
        Arrays.sort(boxTypes, (a,b) -> (a[1] == b[1] ? b[0] - a[0] : b[1] - a[1]));
        // Arrays.sort(boxTypes, (a, b) -> Integer.compare(b[1], a[1]));
        int sum = 0;
        for(int[] box : boxTypes){
            if(box[0] <= truckSize){
                sum += box[0] * box[1];
                truckSize -= box[0];
            }
            else{
                sum += truckSize * box[1];
                break;
            }
        }
        return sum;
    }
}
```
